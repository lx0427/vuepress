# 暴力破解

var twoSum = function(nums, target) {
  // 使用双重循环
  for (let i = 0; i < nums.length; i++) {
    for (let j = 0; j < nums.length; j++) {
      if (nums[i] + nums[j] === target && i !== j) {
        return [i, j]
      }
    }
  }
}

# 需求匹配

空间换时间

var twoSum = function(nums, target) {
  // 相亲: 每个人把自己的需求记录下来
  let obj = {}
  for (let i = 0; i < nums.length; i++) {
    const num = nums[i]
    if (num in obj) {
      return [obj[num], i]
    } else {
      obj[target - num] = i
    }
  }
}
