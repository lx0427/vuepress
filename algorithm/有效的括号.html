# 数组模拟栈

var isValid = function(s) {
  // 有效闭合
  let obj = {
    '(': ')',
    '[': ']',
    '{': '}',
  }
  let stack = []
  for (let i = 0; i < s.length; i++) {
    const ele = s[i]
    if (ele in obj) {
      stack.push(obj[ele])
    } else {
      if (ele !== stack.pop()) return false
    }
  }
  return !stack.length
}
