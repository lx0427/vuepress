### 安装位置

```
|-- D:\android
|   |-- android-studio
|   |-- java
|   |   |-- jdk1.8
|   |   |-- jre1.8
|   |-- sdk // SDK Manager下载安装目录
```

### 环境变量配置

```
## * user

# android studio
ANDROID_HOME: D:\android\sdk
path: 
	%ANDROID_HOME%\platform-tools
	%ANDROID_HOME%\emulator
	%ANDROID_HOME%\tools
	%ANDROID_HOME%\tools\bin
	
# java
JAVA_HOME: D:\android\java\jdk1.8
path:
	%JAVA_HOME%\bin
```
