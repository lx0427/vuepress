# function

(dispatch) => {
  let creators = {
    add: () => ({ type: 'ADD' }),
    minus: () => ({ type: 'MINUS' }),
  }
  // 遍历对象包裹上dispatch
  creators = bindActionCreators(creators, dispatch) // 来自redux
  return { // dispatch和映射方法可同时使用
    dispatch, 
    ...creators,
  }
}
