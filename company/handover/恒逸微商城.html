# 项目结构

|-- dist
|   |-- wx // 微信端打包产出目录
|   |-- app // app端打包产出目录
|-- src
|   |-- components // 自定义组件
|   |-- css // 公用css文件
|   |-- fonts // 字体图标文件
|   |-- i18n // 多语言(弃用)，不可删除，老文件有引用
|   |-- images
|   |-- js
|   |   |-- app // app端，公用js
|   |   |   |-- app.min.js
|   |   |   |-- custom.js
|   |   |-- wx // wx端，公用js
|   |   |   |-- app.min.js
|   |   |   |-- custom.js
|   |   |-- app_config.js // 接口环境配置
|   |   |-- common.js // 全局公用方法
|   |-- libs // 引用第三方js
|   |-- pages // 页面文件，目录文件与页面地址匹配
|-- gulpfile.js // 配置文件
|-- package.json

|-- dev_app // app 本地启动 连接测试环境

|-- dev_app:p // app 本地启动 连接正式环境
|-- test // wx 打包 测试环境
|-- test_app // app 打包 测试环境
|-- build // wx 打包 正式环境
|-- build_app // app 打包 正式环境
